210 if of_type is None: 

211 #Return all arcs 

212 if direction == "pull": 

213 arcs = list(self.in_arcs.values()) 

214 f = 'send_pull_check' 

215 elif direction == "push": 

216 arcs = list(self.out_arcs.values())

240 def get_connected(self, direction = 'pull', of_type = None, tag = 'default'): 

241 """Send push/pull checks to all attached arcs in a given direction 

242 

243 Args: 

244 direction (str, optional): The type of check to send to all attached  

245 nodes. Can be 'push' or 'pull'. The default is 'pull'. 

246 of_type (str or list) : optional, can be specified to send checks only  

247 to nodes of a given type (must be a subclass in nodes.py) 

248 tag (str, optional): optional message to direct query_handler which pull  

249 function to call. Defaults to 'default'. 

250 

251 Returns: 

252 connected (dict) :  

253 Dictionary containing keys: 

254 'avail': (float) - total available volume for push/pull 

255 'priority': (float) - total (availability * preference)  

256 of attached arcs 

257 'allocation': (dict) - contains all attached arcs in specified 

258 direction and respective (availability * preference) 

259  

260 Examples: 

261 >>> vqip_available_to_pull = my_node.get_direction_arcs() 

262 >>> vqip_available_to_push = my_node.get_direction_arcs('push') 

263 >>> avail_reservoir_vqip = my_node.get_direction_arcs('pull', 

264 of_type = 'Reservoir') 

265 >>> avail_sewer_push_to_sewers = my_node.get_direction_arcs('push', 

266 of_type = 'Sewer', 

267 tag = 'Sewer') 

268 """ 

269 

270 #Initialise connected dict 

271 connected = {'avail' : 0, 

272 'priority' : 0, 

273 'allocation' : {}, 

274 'capacity' : {}} 

275 

276 #Get arcs 

277 f, arcs = self.get_direction_arcs(direction, of_type) 

278 

279 #Iterate over arcs, updating connected dict 

280 for arc in arcs: 

281 avail = getattr(arc, f)(tag = tag)['volume'] 

282 if avail < constants.FLOAT_ACCURACY: 

283 avail = 0 #Improves convergence 

284 connected['avail'] += avail 

285 preference = arc.preference 

286 connected['priority'] += avail * preference 

287 connected['allocation'][arc.name] = avail * preference 

288 connected['capacity'][arc.name] = avail 

289 

290 return connected

319 def pull_distributed(self, vqip, of_type = None, tag = 'default'): 

320 """Send pull requests to all (or specified by type) nodes connecting to self. 

321 Iterate until request is met or maximum iterations are hit.  

322 Streamlines if only one in_arc exists. 

323 

324 Args: 

325 vqip (dict): Total amount to pull (by default, only the 

326 'volume' key is used) 

327 of_type (str or list) : optional, can be specified to send checks only  

328 to nodes of a given type (must be a subclass in nodes.py) 

329 tag (str, optional): optional message to direct query_handler which pull  

330 function to call. Defaults to 'default'. 

331 

332 Returns: 

333 pulled (dict): VQIP of combined pulled water 

334 """ 

335 

336 if len(self.in_arcs) == 1: 

337 #If only one in_arc, just pull from that 

338 if of_type == None: 

339 pulled = next(iter(self.in_arcs.values())).send_pull_request(vqip, tag = tag) 

340 elif any([x in of_type for x, y in self.in_arcs_type.items() if len(y) > 0]): 

341 pulled = next(iter(self.in_arcs.values())).send_pull_request(vqip, tag = tag) 

342 else: 

343 #No viable out arcs 

344 pulled = self.empty_vqip() 

345 else: 

346 #Pull in proportion from connected by priority 

347 

348 #Initialise pulled, deficit, connected, iter_ 

349 pulled = self.empty_vqip() 

350 deficit = vqip['volume'] 

351 connected = self.get_connected(direction = 'pull', of_type = of_type, tag = tag) 

352 iter_ = 0 

353 

354 #Iterate over sending nodes until deficit met 

355 while (((deficit > constants.FLOAT_ACCURACY) & 

356 (connected['avail'] > constants.FLOAT_ACCURACY)) & 

357 (iter_ < constants.MAXITER)): 

358 

359 #Pull from connected 

360 for key, allocation in connected['allocation'].items(): 

361 received = self.in_arcs[key].send_pull_request({'volume' : deficit * 

362 allocation / 

363 connected['priority']}, 

364 tag = tag) 

365 pulled = self.sum_vqip(pulled, received) 

366 

367 #Update deficit, connected and iter_ 

368 deficit = vqip['volume'] - pulled['volume'] 

369 connected = self.get_connected(direction = 'pull', of_type = of_type, tag = tag) 

370 iter_ += 1 

371 

372 if iter_ == constants.MAXITER: 

373 print('Maxiter reached in {0} at {1}'.format(self.name, self.t)) 

374 return pulled

376 def push_distributed(self, vqip, of_type = None, tag = 'default'): 

377 """Send push requests to all (or specified by type) nodes connecting to self. 

378 Iterate until request is met or maximum iterations are hit.  

379 Streamlines if only one in_arc exists. 

380 

381 Args: 

382 vqip (dict): Total amount to push 

383 of_type (str or list) : optional, can be specified to send checks only  

384 to nodes of a given type (must be a subclass in nodes.py) 

385 tag (str, optional): optional message to direct query_handler which pull  

386 function to call. Defaults to 'default'. 

387 

388 Returns: 

389 not_pushed_ (dict): VQIP of water that cannot be pushed 

390 """ 

391 

392 if len(self.out_arcs) == 1: 

393 #If only one out_arc, just send the water down that 

394 if of_type == None: 

395 not_pushed_ = next(iter(self.out_arcs.values())).send_push_request(vqip, tag = tag) 

396 elif any([x in of_type for x, y in self.out_arcs_type.items() if len(y) > 0]): 

397 not_pushed_ = next(iter(self.out_arcs.values())).send_push_request(vqip, tag = tag) 

398 else: 

399 #No viable out arcs 

400 not_pushed_ = vqip 

401 else: 

402 #Push in proportion to connected by priority 

403 #Initialise pushed, deficit, connected, iter_ 

404 not_pushed = vqip['volume'] 

405 not_pushed_ = self.copy_vqip(vqip) 

406 connected = self.get_connected(direction = 'push', 

407 of_type = of_type, 

408 tag = tag) 

409 iter_ = 0 

410 if not_pushed > connected['avail']: 

411 #If more water than can be pushed, ignore preference and allocate all available based on capacity 

412 connected['priority'] = connected['avail'] 

413 connected['allocation'] = connected['capacity'] 

414 

415 #Iterate over receiving nodes until sent 

416 while ((not_pushed > constants.FLOAT_ACCURACY) & 

417 (connected['avail'] > constants.FLOAT_ACCURACY) & 

418 (iter_ < constants.MAXITER)): 

419 

420 #Push to connected 

421 amount_to_push = min(connected['avail'], not_pushed) 

422 

423 for key, allocation in connected['allocation'].items(): 

424 

425 to_send = amount_to_push * allocation / connected['priority'] 

426 to_send = self.v_change_vqip(not_pushed_, to_send) 

427 reply = self.out_arcs[key].send_push_request(to_send, tag = tag) 

428 

429 sent = self.extract_vqip(to_send, reply) 

430 not_pushed_ = self.extract_vqip(not_pushed_, sent) 

431 

432 not_pushed = not_pushed_['volume'] 

433 connected = self.get_connected(direction = 'push', 

434 of_type = of_type, 

435 tag = tag) 

436 iter_ += 1 

437 

438 if iter_ == constants.MAXITER: 

439 print('Maxiter reached in {0} at {1}'.format(self.name, self.t)) 

440 

441 return not_pushed_

443 def check_basic(self, direction, vqip = None, of_type = None, tag = 'default'): 

444 """Generic function that conveys a pull or push check onwards to connected  

445 nodes. It is the default behaviour that treats a node like a junction. 

446 

447 Args: 

448 direction (str): can be either 'pull' or 'push' to send checks to  

449 receiving or contributing nodes 

450 vqip (dict, optional): The VQIP to check. Defaults to None (if pulling  

451 this will return available water to pull, if pushing then available  

452 capacity to push). 

453 of_type (str or list) : optional, can be specified to send checks only  

454 to nodes of a given type (must be a subclass in nodes.py) 

455 tag (str, optional): optional message to direct query_handler which pull  

456 function to call. Defaults to 'default'. 

457 

458 Returns: 

459 avail (dict): VQIP responses summed over all requests 

460 """ 

461 f, arcs = self.get_direction_arcs(direction, of_type) 

462 

463 #Iterate over arcs, updating total 

464 avail = self.empty_vqip() 

465 for arc in arcs: 

466 avail = self.sum_vqip(avail, getattr(arc, f)(tag = tag)) 

467 

468 if vqip is not None: 

469 avail = self.v_change_vqip(avail, min(avail['volume'], vqip['volume'])) 

470 

471 return avail
