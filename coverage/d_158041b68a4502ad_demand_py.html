102 'of_type' : 'Land'},

237 def get_house_demand(self): 

238 """Per capita calculations for household wastewater generation. Applies  

239 weighted temperature calculation 

240 

241 Returns: 

242 (dict): A VQIP containg foul water 

243 """ 

244 #TODO water that is consumed but not sent onwards as foul 

245 #Total water required 

246 consumption = self.population * self.per_capita 

247 #Apply pollutants 

248 foul = self.copy_vqip(self.pollutant_load) 

249 #Scale to population 

250 for pol in constants.ADDITIVE_POLLUTANTS: 

251 foul[pol] *= self.population 

252 #Update volume and temperature (which is weighted based on air temperature and constant_temp) 

253 foul['volume'] = consumption 

254 foul['temperature'] = self.get_data_input('temperature') * (1 - self.constant_weighting) + self.constant_temp * self.constant_weighting 

255 return foul
